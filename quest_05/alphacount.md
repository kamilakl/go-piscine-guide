func AlphaCount(s string) int {
	c := 0
	sr := []rune(s)
	for i := 0; i < len(s); i++ {
		for j := 'A'; j <= 'Z'; j++ {
			if sr[i] == j {
				c++
			}
		}
		for j := 'a'; j <= 'z'; j++ {
			if sr[i] == j {
				c++
			}
		}
	}
	return c
}
